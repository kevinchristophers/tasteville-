        e.preventDefault();

        let msg = document.getElementById("feedback").value;

        if(name === "" || msg === ""){
            alert("Please fill all fields");
            return;
        }

            "<p><b>" + name + ":</b> " + msg + "</p>";

        this.reset();
    });
